211inline bool remove_path_suffix(String& p_path, const String& p_suffix) {
212 if (p_suffix.empty()) {
213 return false;
214 }
215 size_t suffix_pos = p_path.find(p_suffix, p_path.find('/'));
216 if (suffix_pos != String::npos) {
217 p_path.erase(suffix_pos, p_suffix.length());
218 return true;
219 } else {
220 return false;
221 }
